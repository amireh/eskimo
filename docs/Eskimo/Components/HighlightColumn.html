Overview

Highlight a particular character of a string with an ASCII arrow.

HighlightColumn.new(line: 0, column: 14) do
  "- include: lol://wut.yml"
end
=> "- include: lol://wut.yml"
   "              ^         "
   "              here      "

def initialize(
  column:,
  line:,
  markers: ['^', 'here'],
  style: [:bold, :red],
  &children
)
  pastel = Pastel.new

  @colorize = ->(str) { pastel.decorate(str, *style) }
  @column = column
  @line = line
  @marker_padding = ' ' * @column
  @markers = markers

  super
end

def render(**)
  lines = super.lines
  line = lines[@line]

  unless line.nil?
    lines[@line] = transform_line!(line, @column, &@colorize)
  end

  lines.join
end
